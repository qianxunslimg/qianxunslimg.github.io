uint16_t CRC16( uint8_t *Frame, uint16_t Len )
{
    uint8_t        ucCRCHi = 0xFF;
    uint8_t        ucCRCLo = 0xFF;
    int                iIndex;

    while( Len-- )
    {
        iIndex = ucCRCLo ^ *( Frame++ );
        ucCRCLo = ( uint8_t )( ucCRCHi ^ CRCHi[iIndex] );
        ucCRCHi = CRCLo[iIndex];
    }
    return ( uint16_t )( ucCRCHi << 8 | ucCRCLo );
}


int main(int argc, char const *argv[]) {
  char a[] = {0x01,0x02,0x03, 0x0d};
  char* temp = a;
  uint16_t c = CRC16((uint8_t *)temp, 4);
  cout<< c <<endl;
  int cL= c%256 & 0xff;   //校验结果的低位
  int cH = c/ 256 & 0xff; //校验结果的高位
}
