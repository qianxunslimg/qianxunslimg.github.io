        vector<vector<int>> ans;

                    left++;

                    ans.push_back({nums[i], nums[left], nums[right]});

            }
        }
        return ans;
    }

class Solution {
public:
    int maxArea(vector<int>& nums) {
      int left = 0, right = nums.size() -1;
      int ans = 0;
      while(left <right){
        int temp = (right-left)*min(nums[left], nums[right]);
        ans = max(ans, temp);
        if(nums[left] > nums[right])
          right--;
        else left++;
      }
      return ans;
    }
};
